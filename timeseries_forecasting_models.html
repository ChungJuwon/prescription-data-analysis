import numpy as np
import tensorflow as tf
from tensorflow.keras.models import Model
from tensorflow.keras.layers import LSTM, Dense, Input
from sklearn.preprocessing import MinMaxScaler
import matplotlib.pyplot as plt

# 한글 폰트 설정
plt.rcParams["font.family"] = "Malgun Gothic"
plt.rcParams["axes.unicode_minus"] = False

# 데이터 불러오기
df_lstm = df_normalized[["이트라코나졸"]].copy()

# Min-Max 스케일링 (LSTM은 정규화된 데이터에서 더 잘 작동함)
scaler = MinMaxScaler()
data_scaled = scaler.fit_transform(df_lstm.values)

# 시퀀스 데이터 생성 함수 (입력: 12개월, 출력: 1개월 예측)
def create_sequences(data, seq_length):
    X, y = [], []
    for i in range(len(data) - seq_length):
        X.append(data[i:i+seq_length])
        y.append(data[i+seq_length])
    return np.array(X), np.array(y)

seq_length = 12  # 12개월 데이터를 기반으로 예측
X, y = create_sequences(data_scaled, seq_length)

# 학습/테스트 데이터 분리
X_train, y_train = X[:-12], y[:-12]  # 마지막 12개월 데이터는 테스트용
X_test, y_test = X[-12:], y[-12:]

# LSTM 모델 구축 (input_shape 대신 Input 사용)
inputs = Input(shape=(seq_length, 1))  #  여기에서 입력 크기 지정
x = LSTM(50, activation="relu", return_sequences=True)(inputs)
x = LSTM(50, activation="relu")(x)
outputs = Dense(1)(x)
model = Model(inputs=inputs, outputs=outputs)  #  Model API 사용

model.compile(optimizer="adam", loss="mse")

# 모델 학습
model.fit(X_train, y_train, epochs=50, batch_size=5, verbose=1)

# 예측 수행
lstm_predictions = model.predict(X_test)  #  변수명 변경
lstm_predictions = scaler.inverse_transform(lstm_predictions)  # 원래 값으로 변환

# 예측된 미래 12개월에 대한 날짜 생성
future_dates = pd.date_range(df_lstm.index[-2] + pd.DateOffset(months=1), periods=12, freq="ME")

# 그래프 그리기
plt.figure(figsize=(12, 6))

# 기존 데이터 (파란색)
plt.plot(df_lstm.index, df_lstm["이트라코나졸"], label="실제 데이터", marker="o")

# LSTM 예측 데이터 (빨간색, 기존 데이터와 겹치지 않음)
plt.plot(future_dates, lstm_predictions, label="LSTM 예측", linestyle="--", marker="o", color="red")

plt.title("LSTM 기반 항진균제(이트라코나졸) 처방 비율 예측")

import numpy as np

# 한글 폰트 설정

# 실제 데이터 로드 (CSV에서 가져오기)

# "Z-score 비율" 컬럼을 실제 값으로 사용
actual_values = actual_data["Z-score 비율"].values  #  기존 df_normalized 대신 CSV 값 사용

# RMSE, MAE 계산 함수
def evaluate_model(actual, predicted):
    mae = mean_absolute_error(actual, predicted)
    rmse = np.sqrt(mean_squared_error(actual, predicted))
    return {"MAE": mae, "RMSE": rmse}

# 각 모델 평가 (예측값을 실제 데이터와 비교)
arima_metrics = evaluate_model(actual_values, arima_predictions.values)
prophet_metrics = evaluate_model(actual_values, prophet_predictions["yhat"].values)
lstm_metrics = evaluate_model(actual_values, lstm_predictions.flatten())

# 결과 데이터프레임으로 정리
results_df = pd.DataFrame([arima_metrics, prophet_metrics, lstm_metrics], 
                          index=["ARIMA", "Prophet", "LSTM"])

# RMSE 값을 기준으로 순위 매기기 (작을수록 좋음)
results_df["순위"] = results_df["RMSE"].rank(method="min").astype(int)

# 결과를 Markdown 표 형태로 출력 (Jupyter Notebook 용)
print("\n모델 성능 비교 결과:\n")
print(results_df.to_markdown())

# **그래프 추가: 기존값, 평가 데이터, 예측 모델 비교**
